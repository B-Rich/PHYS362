\begin{questions}
  \question Consider the operator $\mathcal{A}$, described by the relation
  \[ \mathcal{A}|\psi_0^{\mathrm{SHO}}\rangle =
  |\psi_1^{\mathrm{ISW}}\rangle. \] Here
  $|\psi_0^{\mathrm{SHO}}\rangle$ is the ground state of the harmonic
  oscillator, and $|\psi_1^{\mathrm{ISW}}\rangle$ is the ground state
  of the infinite square well.
  \begin{parts}
    \part[12] Find the eigenvalues and eigenfunctions of
    $\mathcal{A}$.\vspace{\fill}
    \part[4] Is $\mathcal{A}$ hermitian?\vspace{\fill}\newpage
    \part[12] Find the matrix elements of $\mathcal{A}$ in the SHO
    basis.\vspace{\fill}
    \part[12] Find the matrix elements of $\mathcal{A}$ in the ISW
    basis.\vspace{\fill}
  \end{parts}\newpage

  \question
  \begin{parts}
    \part[5] What is the commutator $[T, V]$ for the harmonic
    oscillator?\vspace{\fill}
    \part[5] What is the uncertainty relation for $T$ and $V$ for the
    ground state of the harmonic oscillator?\vspace{\fill}
    \part[10] How do you reconcile this uncertainty with the fact that
    the ground state of the harmonic oscillator is a determinite state
    for energy?\vspace{\fill}
  \end{parts}\newpage

  \question The operator $a_+a_-$ is sometimes called the ``number operator''.
  \begin{parts}
    \part[10] Explain why this name makes sense.\vspace{\fill}
    \part[10] What would you get if you mistakenly used $a_-a_+$
    instead?\vspace{\fill}
    \part[10] Are $\hat H$ and $a_+a_-$ compatible observables?  What
    about $\hat x$ and $a_+a_-$?\vspace{\fill}\newpage
    \part[10] Test the energy-time uncertainty principle for the wave
    function in problem 2.13 and the operator $a_+a_-$.
  \end{parts}
\end{questions}
